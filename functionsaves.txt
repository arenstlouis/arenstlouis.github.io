

artistTest = function (theText) {
			var newtext = document.createElement("h1")
			newtext.innerHTML = theText
			document.body.appendChild(newtext)
		}
getArtistID = function() {
			var request = new XMLHttpRequest();
			
			request.onreadystatechange = function() {
				if (request.readyState == 4)
					if (request.status == 200) {  //successful request OK
					
						console.log(request.responseText)
						
						var res = eval( '(' + request.responseText + ')' )
						
						var count = res['resultsCount']
						var limit = res['resultsLimit']
						
						if (count>limit)
							count = limit
						
						for (var i=0; i<count; i++)
							artistTest(res['artists'][i]['href'])
					}
				}
			var theResource = 'http://http://ws.spotify.com/search/1/artist.json?q=The_Beatles'
			
			request.open('GET', theResource, true)
			
			//request.open('GET', 'urlread.cgi', true)
			
			request.send(null)
			
		}
